二、playbook
1、实现免密登陆
# ansible-doc authorized_key
# vim auth_key.yml
---
- name: configure authorized key
  hosts: all
  tasks:
  - name: root key
    authorized_key:
      user: root
      state: present
      key: "{{ lookup('file', '/root/.ssh/id_rsa.pub') }}"
# ansible-playbook --syntax-check auth_key.yml
# ansible-playbook auth_key.yml -k
# ansible all -m ping

2、配置yum
# mkdir files
# cp /etc/yum.repos.d/server.repo files/
# vim auth_key.yml 追加
  - name: copy yum config file
    copy:
      src: files/server.repo  # 本机目录
      dest: /etc/yum.repos.d/  # 远程目录
# ansible-playbook auth_key.yml

3、配置服务
在web服务器上配置httpd，在数据库服务器上配置mariadb
# vim lamp.yml
---
- name: configure web serivce
  hosts: webservers
  tasks:
    - name: install web app
      yum:
        name: "{{item}}"
        state: present
      with_items:
        - httpd
        - php
        - php-mysql
    - name: config web service
      service:
        name: httpd
        state: started
        enabled: true

- name: configure db service
  hosts: dbservers
  tasks:
    - name: install db app
      yum:
        name: mariadb-server
        state: latest
    - name: config db serivce
      service:
        name: mariadb
        state: started
        enabled: yes
